import { login } from '../services/starwars';
 
export function loginCharacter(username, password) {
console.log('action called',username,password);
  return function (dispatch) {
    console.log('in anonymous');
    login(username).then(response => {
      console.log("response",response.results[0]);
      const {name, birth_year} = response.results[0]
      console.log('name and birht yerar',name,birth_year);
        if(username===response.results[0].name && password===response.results[0].birth_year){
          console.log('login success');
          dispatch({
            type: "LOGIN_SUCCESS"
          });
        }
        else{
          dispatch({
            type: "LOGIN_FAILURE"
          });
        }
      })
      .catch(error => {
        console.log('login error');
        dispatch({
            type: "ERROR"
        });
      });
  };
}
